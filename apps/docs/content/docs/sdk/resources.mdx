import { Accordions, Accordion } from "fumadocs-ui/components/accordion";
import { Callout } from "fumadocs-ui/components/callout";

Chaque ressource est accessible via `client.<resource>`. Cette page détaille toutes les méthodes, leurs paramètres et options.

<Accordions>
  <Accordion title="Invoices — client.invoices">
    | Méthode | Paramètres | Retour | Description |
    |---------|-----------|--------|-------------|
    | `list(params?)` | `{ status?, sorts?, page?, pageSize? }` | `Promise<Invoice[]>` | Lister les factures (paginé) |
    | `listAll(params?)` | `{ status?, sorts?, pageSize? }` | `Promise<Invoice[]>` | Toutes les factures (pagination auto) |
    | `get(id)` | `number` | `Promise<Invoice>` | Détails d'une facture |
    | `create(data)` | `InvoiceCreateParams` | `Promise<Invoice>` | Créer une facture |
    | `update(id, data)` | `number, Partial<InvoiceCreateParams>` | `Promise<Invoice>` | Modifier une facture |
    | `duplicate(id, overrides?)` | `number, { emission_date?, quantity? }` | `Promise<Invoice>` | Dupliquer une facture |
    | `send(id, data)` | `number, { recipients, subject?, message? }` | `Promise<void>` | Envoyer par email |
    | `downloadPdf(id)` | `number` | `Promise<ArrayBuffer>` | Télécharger le PDF |
    | `delete(id)` | `number` | `Promise<void>` | Supprimer un brouillon |

    **InvoiceCreateParams :**

    | Champ | Type | Obligatoire | Description |
    |-------|------|:-----------:|-------------|
    | `emission_date` | `string` | oui | Date d'émission (YYYY-MM-DD) |
    | `lines` | `InvoiceLine[]` | oui | Lignes de la facture |
    | `client` | `{ id: number }` | — | Client existant |
    | `client_name` | `string` | — | Nom du client (si pas de `client`) |
    | `status` | `"draft" \| "saved"` | — | Statut (défaut: `draft`) |
    | `title` | `string` | — | Titre |
    | `free_field` | `string` | — | Champ libre |
    | `due_date` | `string` | — | Date d'échéance |
    | `template` | `string` | — | Template de facture |

    **Filtres `list()` :**

    | Champ | Description |
    |-------|-------------|
    | `status` | `draft`, `saved`, `sent`, `paid` |
    | `sorts` | Ex: `invoice_number:desc` (défaut) |
    | `page` | Numéro de page (défaut: 1) |
    | `pageSize` | Résultats par page (défaut: 25) |
  </Accordion>

  </Accordion>

  <Accordion title="Clients — client.clients">
    | Méthode | Paramètres | Retour | Description |

    | Champ | Type | Obligatoire | Description |
    |-------|------|:-----------:|-------------|
    | `name` | `string` | oui | Nom du client |
    | `email` | `string` | — | Email |
    | `phone` | `string` | — | Téléphone |
    | `address` | `string` | — | Adresse |
    | `postal_code` | `string` | — | Code postal |
    | `city` | `string` | — | Ville |
    | `siren_or_siret` | `string` | — | SIREN ou SIRET |
    | `professional` | `boolean` | — | Client professionnel (défaut: `true`) |
  </Accordion>

  <Accordion title="BankAccounts — client.bankAccounts">

  <Accordion title="BankTransactions — client.bankTransactions">
    | Méthode | Paramètres | Retour | Description |
    |---------|-----------|--------|-------------|
    | `list(params?)` | `BankTransactionsListParams` | `Promise<BankTransactionsResponse>` | Lister (paginé) |
    | `listAll(params?)` | même sans `page` | `Promise<BankTransaction[]>` | Toutes les transactions (pagination auto) |
    | `get(id)` | `number` | `Promise<BankTransaction>` | Détails d'une transaction |
    | `unimputed()` | — | `Promise<BankTransaction[]>` | Transactions non imputées |
    | `labelSuggestions(id)` | `number` | `Promise<LabelSuggestion[]>` | Suggestions de label |
    | `impute(id, imputations)` | `number, ImputationParams[]` | `Promise<BankTransaction>` | Imputer une transaction |

    **BankTransactionsListParams :**

    | Champ | Type | Défaut | Description |
    |-------|------|--------|-------------|
    | `from` | `string` | — | Date de début (YYYY-MM-DD) |
    | `to` | `string` | — | Date de fin (YYYY-MM-DD) |
    | `search` | `string` | — | Recherche dans le libellé |
    | `bank_account` | `number` | — | ID du compte bancaire |
    | `hide_refused` | `boolean` | `false` | Masquer les refusées |
    | `amount_type` | `"positive" \| "negative"` | — | Filtrer par type |
    | `sorts` | `string` | — | Tri (ex: `date:desc`) |
    | `page` | `number` | `1` | Numéro de page |
    | `pageSize` | `number` | `100` | Résultats par page |

    **ImputationParams :**

    | Champ | Type | Description |
    |-------|------|-------------|
    | `label` | `{ id: number }` | Label comptable |
    | `amount` | `number` | Montant imputé |
    | `documents` | `array` | Documents associés |
    | `accountant_detail_requests` | `array` | Demandes de détails |
  </Accordion>

  <Accordion title="Documents — client.documents">
    | Méthode | Paramètres | Retour | Description |

    | Méthode | Paramètres | Retour | Description |
    |---------|-----------|--------|-------------|

<Callout type="info" title="Pagination automatique">
  Les méthodes `listAll()` (disponibles sur `invoices` et `bankTransactions`) itèrent automatiquement sur toutes les pages. Pour les autres ressources, la pagination utilise le header HTTP `Range` et retourne toutes les données en une seule requête.
</Callout>
